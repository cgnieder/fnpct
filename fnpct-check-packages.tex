\long\def\StartPackageCheck#1#2\EndPackageCheck{%
  \def\thispackagename{#1}%
  \ifx\testpackagename\thispackagename
  #2%
  \else
  \fi
}

  \setfnpct{dont-mess-around}%
  \IfBooleanTF{#1}{%
    \IfNoValueTF{#2}
      {#NOTE{#4}}
      {\IfNoValueTF{#3}{#NOTE[#2]{#4}}{#NOTE[#2][#3]{#4}}}%
  }{%
  \IfNoValueTF{#2}
    {#NOTE*{#4}}
    {\IfNoValueTF{#3}{#NOTE*[#2]{#4}}{#NOTE*[#2][#3]{#4}}}%
  }%
}

\footcite{knuth:ct:c}\footcite{knuth:ct:d}.
